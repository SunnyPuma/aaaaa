JS代码

  1. //获得选中的标签页索引
  2. var tab = $('#tt').tabs('getSelected');
  3. var index = $('#tt').tabs('getTabIndex', tab);
  4. $.messager.alert("提示信息", "你选中的标签页索引是：" + index);
  5. //选中标签页
  6. $('#tt').tabs('select', 2);
  7. $('#tt').tabs('select', 'Tab3');
