示例

 启用 禁用 设置一周的第一天 设置值 获取值

HTML代码

  1. <input id="dd" data-options="editable:false" class="easyui-datebox"/>

JS代码

  1. //启用
  2. $('#dd').datebox('enable');
  4. //禁用 class="alt"
  5. $('#dd').datebox('disable');
  7. //设置一周的第一天是星期几
  8. var c = $('#dd').datebox('calendar'); //获取日历对象
  9. c.calendar({
  10.     firstDay: 1 //0是周日，1是周一，...
  11. });
  13. //设置值
  14. $('#dd').datebox('setValue', '2014-05-01');
  16. //获取值
  17. $('#dd').datebox('getValue');
