• 普通弹出框
  • 大小固定
  • 带工具栏
  • 带按钮栏
  • 模态对话框

示例

普通弹出框

HTML代码

  1. <div id="win" class="easyui-window" title="My Window"></div>

JS代码

  1. $(function(){
  2.     $('#win').window({
  3.         width:600,
  4.         height:400,
  5.         closed: true,
  6.     });
  7. });
  8. //打开对话框
  9. $('#win').window('open');

带工具栏

HTML代码

  1. <div id="win" class="easyui-dialog" title="My Dialog"></div>

JS代码

  1. $(function(){
  2.     $('#win').dialog({
  3.         width: 600,
  4.         height: 400,
  5.         closed: true,
  6.         toolbar: [{
  7.             text: '编辑',
  8.             iconCls: 'icon-edit',
  9.             handler: function(){
  10.                 alert('edit');
  11.             }
  12.         }, {
  13.             text: '帮助',
  14.             iconCls: 'icon-help',
  15.             handler: function(){
  16.                 alert('help');
  17.             }
  18.         }]
  19.     });
  20. });
  21. //打开对话框
  22. $('#win').dialog('open');

示例

带按钮栏

HTML代码

  1. <div id="win" class="easyui-dialog" title="My Dialog"></div>

JS代码

  1. $(function(){
  2.     $('#win').dialog({
  3.         width: 600,
  4.         height: 400,
  5.         closed: true,
  6.         buttons: [{
  7.             text: '确定',
  8.             iconCls: 'icon-ok',
  9.             handler: function(){
  10.             }
  11.         }, {
  12.             text: '关闭',
  13.             iconCls: 'icon-cancel',
  14.             handler: function(){
  15.                 $('#win').dialog('close');
  16.             }
  17.         }]
  18.     });    
  19. });
  20. //打开对话框
  21. $('#win').dialog('open');
